CREATE DABATABSE IF NOT EXISTS ampoule;

DROP TABLE IF EXISTS ampoule;
CREATE TABLE IF NOT EXISTS ampoule (
    id INT(11) NOT NULL AUTO_INCREMENT,
    date_changement DATETIME DEFAULT CURRENT_TIMESTAMP NOT NULL,
    etage INT(2) NOT NULL,
    position INT(1) NOT NULL,
    prix DECIMAL(9,2) NOT NULL,
    PRIMARY KEY (id));

DROP TABLE IF EXISTS gardien;
CREATE TABLE gardien (
    id INT(11) NOT NULL AUTO_INCREMENT,
    name VARCHAR(11) NOT NULL,
    username VARCHAR(255) NOT NULL,
    password VARCHAR(255) NOT NULL,
    PRIMARY KEY (id)
    );
